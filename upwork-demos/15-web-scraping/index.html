import requests
from bs4 import BeautifulSoup
from dataclasses import dataclass
from typing import List, Optional
import time
import json
import csv
from tenacity import retry, stop_after_attempt, wait_exponential

@dataclass
class Product:
    """Structured product data"""
    name: str
    price: float
    rating: Optional[float]
    reviews_count: int
    url: str
    in_stock: bool

class ProductScraper:
    """
    Production-ready web scraper with:
    - Automatic retries with exponential backoff
    - Rate limiting to respect server
    - Pagination handling
    - Clean data output
    """
    
    def __init__(self, base_url: str, delay: float = 1.0):
        self.base_url = base_url
        self.delay = delay
        self.session = requests.Session()
        self.session.headers.update({
            'User-Agent': 'Mozilla/5.0 (compatible; DataBot/1.0)',
            'Accept': 'text/html,application/xhtml+xml'
        })
    
    @retry(stop=stop_after_attempt(3), wait=wait_exponential(multiplier=1, max=10))
    def fetch_page(self, url: str) -> BeautifulSoup:
        """Fetch and parse a page with automatic retry"""
        response = self.session.get(url, timeout=30)
        response.raise_for_status()
        time.sleep(self.delay)  # Rate limiting
        return BeautifulSoup(response.text, 'html.parser')
    
    def parse_product(self, element) -> Product:
        """Extract product data from HTML element"""
        name = element.select_one('.product-title').text.strip()
        
        # Parse price (handle currency symbols)
        price_text = element.select_one('.price').text
        price = float(price_text.replace('$', '').replace(',', ''))
        
        # Parse rating (may not exist)
        rating_el = element.select_one('.rating')
        rating = float(rating_el.get('data-rating')) if rating_el else None
        
        # Parse review count
        reviews_el = element.select_one('.review-count')
        reviews = int(reviews_el.text.split()[0]) if reviews_el else 0
        
        # Get URL and stock status
        url = element.select_one('a')['href']
        in_stock = 'out-of-stock' not in element.get('class', [])
        
        return Product(
            name=name,
            price=price,
            rating=rating,
            reviews_count=reviews,
            url=url if url.startswith('http') else f"{self.base_url}{url}",
            in_stock=in_stock
        )
    
    def scrape_category(self, category_url: str, max_pages: int = 10) -> List[Product]:
        """Scrape all products from a category with pagination"""
        products = []
        page = 1
        
        while page <= max_pages:
            url = f"{category_url}?page={page}"
            print(f"Scraping page {page}...")
            
            try:
                soup = self.fetch_page(url)
                product_elements = soup.select('.product-card')
                
                if not product_elements:
                    print("No more products found")
                    break
                
                for el in product_elements:
                    try:
                        product = self.parse_product(el)
                        products.append(product)
                    except Exception as e:
                        print(f"Error parsing product: {e}")
                        continue
                
                page += 1
                
            except Exception as e:
                print(f"Error fetching page {page}: {e}")
                break
        
        print(f"Scraped {len(products)} products")
        return products
    
    def export_json(self, products: List[Product], filename: str):
        """Export products to JSON file"""
        data = [p.__dict__ for p in products]
        with open(filename, 'w') as f:
            json.dump(data, f, indent=2)
        print(f"Exported to {filename}")
    
    def export_csv(self, products: List[Product], filename: str):
        """Export products to CSV file"""
        with open(filename, 'w', newline='') as f:
            writer = csv.DictWriter(f, fieldnames=Product.__annotations__.keys())
            writer.writeheader()
            for p in products:
                writer.writerow(p.__dict__)
        print(f"Exported to {filename}")


# Usage example
if __name__ == '__main__':
    scraper = ProductScraper(
        base_url='https://example-store.com',
        delay=1.5  # 1.5 seconds between requests
    )
    
    products = scraper.scrape_category(
        category_url='https://example-store.com/electronics',
        max_pages=5
    )
    
    # Export in multiple formats
    scraper.export_json(products, 'products.json')
    scraper.export_csv(products, 'products.csv')
