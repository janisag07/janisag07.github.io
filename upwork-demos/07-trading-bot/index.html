🐍 Python - Binance Integration

Python 3.10+ 
import asyncio
from binance import AsyncClient, BinanceSocketManager
from telegram import Bot
import os

# Configuration
BINANCE_API_KEY = os.getenv('BINANCE_API_KEY')
BINANCE_SECRET = os.getenv('BINANCE_SECRET')
TELEGRAM_TOKEN = os.getenv('TELEGRAM_BOT_TOKEN')
CHAT_ID = os.getenv('TELEGRAM_CHAT_ID')

class TradingBot:
    def __init__(self):
        self.telegram = Bot(token=TELEGRAM_TOKEN)
        self.price_alerts = {}  # symbol -> {above: price, below: price}
        
    async def start(self):
        # Initialize Binance client
        self.client = await AsyncClient.create(
            BINANCE_API_KEY, 
            BINANCE_SECRET
        )
        self.bsm = BinanceSocketManager(self.client)
        
        # Start listening to price stream
        await self.listen_prices(['BTCUSDT', 'ETHUSDT'])
    
    async def listen_prices(self, symbols):
        """Listen to real-time price updates via WebSocket"""
        streams = [f"{s.lower()}@ticker" for s in symbols]
        socket = self.bsm.multiplex_socket(streams)
        
        async with socket as stream:
            while True:
                msg = await stream.recv()
                await self.process_ticker(msg)
    
    async def process_ticker(self, msg):
        """Process incoming ticker and check alerts"""
        data = msg['data']
        symbol = data['s']
        price = float(data['c'])  # Current price
        change = float(data['P'])  # 24h change %
        
        # Check price alerts
        if symbol in self.price_alerts:
            alert = self.price_alerts[symbol]
            if 'above' in alert and price >= alert['above']:
                await self.send_alert(symbol, price, 'above')
            if 'below' in alert and price <= alert['below']:
                await self.send_alert(symbol, price, 'below')
    
    async def send_alert(self, symbol, price, direction):
        """Send Telegram notification"""
        emoji = '🚀' if direction == 'above' else '📉'
        message = f"""
{emoji} Price Alert: {symbol}

Price: ${price:,.2f}
Triggered: {direction} threshold

Sent by TradingBot
"""
        await self.telegram.send_message(
            chat_id=CHAT_ID,
            text=message,
            parse_mode='HTML'
        )

# Run the bot
if __name__ == '__main__':
    bot = TradingBot()
    bot.price_alerts = {
        'BTCUSDT': {'above': 100000, 'below': 90000},
        'ETHUSDT': {'above': 4000, 'below': 3000}
    }
    asyncio.run(bot.start())

📦 Node.js - Coinbase Integration

TypeScript 
import { CoinbaseClient } from '@coinbase/coinbase-sdk';
import { Telegraf } from 'telegraf';
import WebSocket from 'ws';

interface PriceAlert {
  symbol: string;
  above?: number;
  below?: number;
}

class CoinbaseTradingBot {
  private bot: Telegraf;
  private alerts: Map<string, PriceAlert> = new Map();
  private chatId: string;
  
  constructor() {
    this.bot = new Telegraf(process.env.TELEGRAM_BOT_TOKEN!);
    this.chatId = process.env.TELEGRAM_CHAT_ID!;
    this.setupCommands();
  }
  
  setupCommands() {
    // /price BTC - Get current price
    this.bot.command('price', async (ctx) => {
      const symbol = ctx.message.text.split(' ')[1] || 'BTC';
      const price = await this.getPrice(symbol);
      ctx.reply(`💰 ${symbol}: $${price.toLocaleString()}`);
    });
    
    // /alert BTC above 100000
    this.bot.command('alert', (ctx) => {
      const [, symbol, direction, priceStr] = ctx.message.text.split(' ');
      const price = parseFloat(priceStr);
      
      this.alerts.set(symbol, { 
        symbol, 
        [direction]: price 
      });
      
      ctx.reply(`✅ Alert set: ${symbol} ${direction} $${price.toLocaleString()}`);
    });
  }
  
  async getPrice(symbol: string): Promise<number> {
    const response = await fetch(
      `https://api.coinbase.com/v2/prices/${symbol}-USD/spot`
    );
    const data = await response.json();
    return parseFloat(data.data.amount);
  }
  
  startPriceStream() {
    const ws = new WebSocket('wss://ws-feed.exchange.coinbase.com');
    
    ws.on('open', () => {
      ws.send(JSON.stringify({
        type: 'subscribe',
        channels: [{ name: 'ticker', product_ids: ['BTC-USD', 'ETH-USD'] }]
      }));
    });
    
    ws.on('message', async (data) => {
      const msg = JSON.parse(data.toString());
      if (msg.type === 'ticker') {
        await this.checkAlerts(msg.product_id, parseFloat(msg.price));
      }
    });
  }
  
  async checkAlerts(productId: string, price: number) {
    const symbol = productId.replace('-USD', '');
    const alert = this.alerts.get(symbol);
    
    if (!alert) return;
    
    if (alert.above && price >= alert.above) {
      await this.sendAlert(symbol, price, '📈 Above');
      this.alerts.delete(symbol);
    }
    if (alert.below && price <= alert.below) {
      await this.sendAlert(symbol, price, '📉 Below');
      this.alerts.delete(symbol);
    }
  }
  
  async sendAlert(symbol: string, price: number, type: string) {
    await this.bot.telegram.sendMessage(
      this.chatId,
      `🚨 *${type} Alert*\n\n${symbol}: $${price.toLocaleString()}`,
      { parse_mode: 'Markdown' }
    );
  }
  
  async start() {
    this.startPriceStream();
    await this.bot.launch();
    console.log('🤖 Trading bot started!');
  }
}

const bot = new CoinbaseTradingBot();
bot.start();
