📋 Project: Invoice Management Tool

A simple business tool for creating, sending, and tracking invoices. Demonstrates full-stack TypeScript with shared types, REST API, authentication, and real-time updates.

Shared Types (used by both frontend & backend)

packages/shared/types/invoice.ts 
// Shared types ensure type safety across the entire stack

export interface Invoice {
  id: string;
  number: string;
  clientId: string;
  client: Client;
  items: InvoiceItem[];
  status: InvoiceStatus;
  dueDate: Date;
  total: number;
  createdAt: Date;
  paidAt?: Date;
}

export interface InvoiceItem {
  description: string;
  quantity: number;
  unitPrice: number;
  total: number;
}

export type InvoiceStatus = 'draft' | 'sent' | 'paid' | 'overdue';

export interface CreateInvoiceDto {
  clientId: string;
  items: Omit<InvoiceItem, 'total'>[];
  dueDate: string;
}

NestJS Backend - Invoice Controller

apps/api/src/invoices/invoices.controller.ts 
import { Controller, Get, Post, Body, Param, UseGuards } from '@nestjs/common';
import { Invoice, CreateInvoiceDto } from '@shared/types';
import { InvoicesService } from './invoices.service';
import { JwtAuthGuard } from '../auth/jwt-auth.guard';
import { CurrentUser } from '../auth/current-user.decorator';

@Controller('invoices')
@UseGuards(JwtAuthGuard)
export class InvoicesController {
  constructor(private invoicesService: InvoicesService) {}

  @Get()
  async findAll(@CurrentUser() user: User): Promise<Invoice[]> {
    return this.invoicesService.findByUser(user.id);
  }

  @Get(':id')
  async findOne(@Param('id') id: string): Promise<Invoice> {
    return this.invoicesService.findById(id);
  }

  @Post()
  async create(
    @CurrentUser() user: User,
    @Body() dto: CreateInvoiceDto
  ): Promise<Invoice> {
    return this.invoicesService.create(user.id, dto);
  }

  @Post(':id/send')
  async send(@Param('id') id: string): Promise<Invoice> {
    return this.invoicesService.sendToClient(id);
  }
}

NestJS Backend - Invoice Service

apps/api/src/invoices/invoices.service.ts 
import { Injectable } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';
import { EmailService } from '../email/email.service';
import { CreateInvoiceDto, Invoice } from '@shared/types';

@Injectable()
export class InvoicesService {
  constructor(
    private prisma: PrismaService,
    private emailService: EmailService
  ) {}

  async create(userId: string, dto: CreateInvoiceDto): Promise<Invoice> {
    // Calculate totals
    const items = dto.items.map(item => ({
      ...item,
      total: item.quantity * item.unitPrice
    }));
    const total = items.reduce((sum, item) => sum + item.total, 0);

    // Generate invoice number
    const number = await this.generateInvoiceNumber(userId);

    return this.prisma.invoice.create({
      data: {
        number,
        userId,
        clientId: dto.clientId,
        items: { create: items },
        total,
        dueDate: new Date(dto.dueDate),
        status: 'draft'
      },
      include: { client: true, items: true }
    });
  }

  async sendToClient(id: string): Promise<Invoice> {
    const invoice = await this.prisma.invoice.update({
      where: { id },
      data: { status: 'sent' },
      include: { client: true }
    });

    // Send email with invoice PDF
    await this.emailService.sendInvoice(invoice);

    return invoice;
  }
}
